    }

    private static String FunctionCharBinary(String name) {
        String result = "";
        char ch = ' ';

        for (int i = 0; i < name.length(); i++) {
            ch = name.charAt(0);
            result = Integer.toBinaryString(ch) + " ";
        }

        return result;
    }

    private static void FunctionCharInt() {
        int Ncol = 3;
        int iCol=256/Ncol;
        for (int i = 0; i < 255; i++) {
            System.out.println(i + "\t" + Integer.toString(i, 16)
                    + " =" + (char) i);
            System.out.print((i + iCol * 1) + " - " + (char) (i + iCol * 1)+"\t");
            System.out.print("\t" + (i + iCol * 2) + " - " + (char) (i + iCol * 2)+"\t");
